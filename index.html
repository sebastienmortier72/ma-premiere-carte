// On initialise la carte (si ce n'est pas déjà fait) var map = L.map('map').setView([44.7, 5.0], 9); L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map); // On charge le résultat du robot fetch('resultat.geojson') .then(res => { if (!res.ok) throw new Error("Fichier resultat.geojson introuvable"); return res.json(); }) .then(data => { L.geoJSON(data, { style: { color: "#3498db", weight: 2, fillOpacity: 0.3 }, onEachFeature: function (feature, layer) { // On cherche le nom de la commune peu importe la casse let n = feature.properties; let nom = n.NOM || n.nom || n.NOM_COMM || n.nom_comm || "Commune"; let status = n.status_robot || "En attente..."; layer.bindPopup("" + nom + "
Status : " + status); } }).addTo(map); }) .catch(err => console.error("Erreur Leaflet :", err));
